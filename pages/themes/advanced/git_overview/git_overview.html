Open a Terminal/Command Prompt


  			# Create your project directory
  			$ mkdir my_project

  			# go into it
  			$ cd my_project

  			# initialize the new repository
  			$ git init

  			# do some work and save your files
  			$ echo "I'm working hard..." > file1.txt

  			# stage all your work
  			$ git add -A

  			# commit your work
  			$ git commit -am'Work 1 Done'
